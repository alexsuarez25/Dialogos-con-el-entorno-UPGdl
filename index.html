  const onU=useCallback(()=>{dr.current.a=false;},[]);

  useEffect(()=>{const el=svgR.current;if(!el)return;el.addEventListener('wheel',onW,{passive:false});return()=>el.removeEventListener('wheel',onW);},[onW]);


  const pN=sM?.key==="proyectos"?projPos():[];

  const cS=sM?.key==="cinov"?subPos("cinov",CINOV_SUBS):[];

  const aS=sM?.key==="aliados"?subPos("aliados",ALIADOS_SUBS):[];

  const iSub=sM?.key==="investigacion"?subPos("investigacion",INVEST_SUBS):[];

  const allS=[...cS,...aS,...iSub];

  const sItems=(sSub&&!sSub.listMode)?itemPos(sSub,sSub.items||[]):[];

  const tN=sProj?temaPos(sProj):[];

  const vS=vb.map(v=>Math.round(v)).join(" ");


  return(

        <div onClick={goHome} style={{width:34,height:34,borderRadius:"50%",cursor:"pointer",background:C.gold,display:"flex",alignItems:"center",justifyContent:"center"}}>

        <div style={{fontSize:14,fontWeight:800,color:C.textDk}}>Diálogos con el Entorno</div>

        {sM && <button onClick={goHome} style={{marginLeft:"auto",background:"none",border:`1px solid ${C.gold}`,borderRadius:5,padding:"5px 10px",cursor:"pointer"}}>Volver al Inicio</button>}

      </div>


      <div ref={svgR} style={{position:"absolute",inset:0,cursor:dr.current.a?"grabbing":"grab"}}

          onClick={()=>!dr.current.m && (sItem?setSItem(null):null)} onMouseDown={onD} onMouseMove={onM2} onMouseUp={onU}>

        <svg viewBox={vS} style={{width:"100%",height:"100%"}}>

          <rect x="0" y="0" width={WW} height={HH} fill={C.bg}/>

          {/* Central logo y circulos */}

          <circle cx={CX} cy={CY} r={NR0} fill={C.gold} onClick={goHome} style={{cursor:"pointer"}}/>

          <image href={LOGO} x={CX-100} y={CY-100} width={200} height={200} style={{pointerEvents:"none"}}/>


          {/* MAIN NODES */}

          {MAIN.map(m=>{

            const o=mPos[m.key];

            return(<g key={m.key} onClick={(e)=>{e.stopPropagation();goMain(m)}} style={{cursor:"pointer"}}>

              <circle cx={o.x} cy={o.y} r={NR1} fill={m.color}/>

              <MT text={m.name} x={o.x} y={o.y} size={28} fill={C.white}/>

            </g>);

          })}


          {/* SUBS */}

          {allS.map((s,i)=>(<g key={i} onClick={(e)=>{e.stopPropagation();goSub(s)}} style={{cursor:"pointer"}}>

            <circle cx={s.x} cy={s.y} r={NR2} fill={C.white} stroke={s.color} strokeWidth={3}/>

          </g>))}

          {pN.map((p,i)=>(<g key={i} onClick={(e)=>{e.stopPropagation();goProj(p)}} style={{cursor:"pointer"}}>

          </g>))}

          {sItems.map((it,i)=>(<g key={i} onClick={(e)=>{e.stopPropagation();setSItem(it);flyTo(camN(it.x,it.y),700)}} style={{cursor:"pointer"}}>

            <circle cx={it.x} cy={it.y} r={NR3} fill={C.white} stroke={sSub.color} strokeWidth={2}/>

          {tN.map((t,i)=>(<g key={i} onClick={(e)=>{e.stopPropagation();setSTema(t)}} style={{cursor:"pointer"}}>

        </svg>

      {panel && sProj && <PanelC data={sProj.desc} accent={sProj.color} onClose={()=>setPanel(false)} isCentral={false}/>}

      {panel && sSub?.listMode && (

        <div style={{position:"absolute",top:0,right:0,width:400,height:"100%",background:C.panelBg,zIndex:30,padding:20,borderLeft:`2px solid ${sSub.color}`}}>

          </div>

          <button onClick={()=>setPanel(false)}>Cerrar</button>

        </div>

      )}


      {sItem && <ItemPop item={sItem} onClose={()=>setSItem(null)} noteText={nt} setNoteText={setNt} onSave={()=>saveN(sItem.name,nt)} noteHistory={notes[sItem.name]||[]} onDelete={(idx)=>delNote(sItem.name,idx)} />}

    </div>

  );
